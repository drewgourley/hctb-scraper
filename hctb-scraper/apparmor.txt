#include <tunables/global>

profile hctb-scraper flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Allow reading the Node.js binary and scripts
  /usr/bin/node rix,
  /usr/local/bin/node rix,
  /usr/bin/hctb-scraper rix,
  /data/** rw, # Read/write access to persistent data/config
  /tmp/** rw, # Temporary files

  # Allow reading system files needed for basic operation
  /etc/passwd r,
  /etc/group r,

  # Allow access to device files for logging/output
  /dev/tty rw,

  # Allow network access
  network inet stream,
  network inet dgram,

  # Deny everything else by default
  deny /** w,
}
